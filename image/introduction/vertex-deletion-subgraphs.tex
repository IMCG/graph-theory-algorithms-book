%% graph G
\subfigure[$G$]{
\begin{tikzpicture}
[lineDecorate/.style={-,thick},%
  nodeDecorate/.style={shape=circle,inner sep=2pt,draw,thick},%
  scale=1.5]
%% nodes or vertices
\foreach \nodename/\x/\y in {
  a/0/1, b/1/1, c/1/2, d/1/0, 1/2/1.5, 2/2/0.5, 3/3/1.5, 4/3/0.5, e/4/1}
{
  \node (\nodename) at (\x,\y) [nodeDecorate] {};
}
\foreach \nodename/\direction/\navigate in {
  a/left/west, b/above left/west, c/above/north, d/below/south, e/right/east}
{
  \node [\direction] at (\nodename.\navigate) {\footnotesize$\nodename$};
}
%% edges or lines
\path
\foreach \startnode/\endnode in {
  a/b, a/c, a/d, b/c, b/d, b/1, b/2, c/1, c/2, d/1, d/2, 1/2, 1/3,
  1/4, 2/3, 2/4, 3/4, 3/e, 4/e}
{
  (\startnode) edge[lineDecorate] node {} (\endnode)
};
\end{tikzpicture}
}
