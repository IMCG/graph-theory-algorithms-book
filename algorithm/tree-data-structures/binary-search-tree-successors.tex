\DontPrintSemicolon
\SetAlgoNoLine
%%
%% data section
\SetKwData{NULL}{\footnotesize{NULL}}
%%
%% input
\KwIn{A nonempty binary search tree $T$ and a vertex $v$ that is not
  a maximum of $T$.}
%%
%% output
\KwOut{The successor of $v$.}
\BlankLine
%%
%% algorithm body
\If{$\rightChild[v] \neq \NULL$}{
  \Return minimum vertex of $v$'s right subtree as per Algorithm~\ref{alg:tree_data_structures:binary_search_tree_minimum_key}\;
}
$u \assign \parent(v)$\;
\While{\rm $u \neq \NULL$ and $v = \rightChild[u]$}{
  $v \assign u$\;
  $u \assign \parent(u)$\;
}
\Return $u$\;
