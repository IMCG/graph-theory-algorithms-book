\DontPrintSemicolon
\SetAlgoNoLine
%%
%% input
\KwIn{A finite list $L$ of $n > 0$ elements on which a total order is
  defined.}
%%
%% output
\KwOut{The same list $L$ sorted by the total order relation defined
  on its elements.}
\BlankLine
%%
%% algorithm body
$Q \assign [\,]$\;
\For{$i \assign 1, 2, \dots, n$}{
  $e \assign \dequeue(L)$\;
  $\insertElem(Q, e, e)$\;
}
\For{$i \assign 1, 2, \dots, n$}{
  $e \assign \extractMin(Q)$\;
  $\enqueue(L, e)$\;
}
